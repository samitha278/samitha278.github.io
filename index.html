Efficient KV-Cache Pruning

A minimalist implementation of attention-aware KV-cache eviction for VLMs

vlm_kv_pruner.py Python · PyTorch 
1  # KV-Cache Pruning via Cross-Attention Head Importance Scoring
2  # Samitha Sahan Bandara · CVPR 2024
3
4  import  torch
5  import  torch.nn as  nn
6  from  torch import  Tensor
7
8
9  @torch.no_grad()
10 def compute_head_importance(
11     attn_weights: Tensor,     # (B, H, T_q, T_kv)
12     top_k: int = 4
13 ) -> Tensor:
14     """Rank attention heads by cross-modal saliency score."""
15     # Mean attention entropy across query positions
16     entropy = -(attn_weights * (attn_weights + 1e-9).log()).sum(-1)
17     importance = entropy.mean(-1)              # (B, H)
18     return  importance.topk(top_k, dim=-1).indices  # (B, k)
19
20
21 class AdaptiveKVCache(nn.Module):
22     def __init__(self, max_len: int, budget: float = 0.5):
23         super().__init__()
24         self.max_len   = max_len
25         self.budget    = budget     # fraction of tokens to keep
26         self.cache_k: Tensor | None = None
27         self.cache_v: Tensor | None = None
28
29     def update(self, k: Tensor, v: Tensor, scores: Tensor) -> tuple[Tensor, Tensor]:
30         """Append new KV pairs and evict low-importance tokens."""
31         self.cache_k = k if self.cache_k is None else torch.cat([self.cache_k, k], dim=-2)
32         self.cache_v = v if self.cache_v is None else torch.cat([self.cache_v, v], dim=-2)
33
34         if  self.cache_k.size(-2) > self.max_len:
35             keep = int(self.max_len * self.budget)
36             idx  = scores.topk(keep, dim=-1).indices.sort().values
37             self.cache_k = self.cache_k.index_select(-2, idx)
38             self.cache_v = self.cache_v.index_select(-2, idx)
39
40         return  self.cache_k, self.cache_v
